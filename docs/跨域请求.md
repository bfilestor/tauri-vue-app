```java

package com.example.config;

import java.util.List;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.CorsConfigurationSource;
import org.springframework.web.cors.UrlBasedCorsConfigurationSource;
import org.springframework.security.config.Customizer;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.web.SecurityFilterChain;

@Configuration
public class CorsConfig {

    @Bean
    public CorsConfigurationSource corsConfigurationSource() {
        CorsConfiguration c = new CorsConfiguration();

        // 按需保留你实际会用到的 Origin
        c.setAllowedOriginPatterns(List.of(
            "http://localhost:1420",
            "tauri://localhost",
            "https://tauri.localhost"
        ));

        c.setAllowedMethods(List.of("GET", "POST", "PUT", "DELETE", "PATCH", "OPTIONS"));
        c.setAllowedHeaders(List.of(
            "Authorization",
            "Content-Type",
            "X-Trace-Id",
            "X-Client-Id",
            "X-Device-Id",
            "X-Timestamp",
            "X-Nonce",
            "X-Signature",
            "X-App-Version",
            "Idempotency-Key"
        ));
        c.setExposedHeaders(List.of("Authorization", "X-Trace-Id"));
        c.setAllowCredentials(true); // 若不用 Cookie，可改 false
        c.setMaxAge(3600L);

        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();

        // 如果你配置了 server.servlet.context-path=/health，这里用 /app-api/**
        source.registerCorsConfiguration("/app-api/**", c);

        // 如果你没配 context-path，而接口就是 /health/app-api/**，改成这行：
        // source.registerCorsConfiguration("/health/app-api/**", c);

        return source;
    }

    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
        http
            .cors(Customizer.withDefaults())
            .csrf(csrf -> csrf.disable());
        return http.build();
    }
}

nginx 

``` 
# 放在 http {} 里
map $http_origin $cors_origin {
    default "";
    "~^http://localhost:1420$" $http_origin;
    "~^tauri://localhost$" $http_origin;
    "~^https://tauri\.localhost$" $http_origin;
}

server {
    listen 80;
    server_name 192.168.1.75;

    location /health/app-api/ {
        # 预检请求
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin $cors_origin always;
            add_header Access-Control-Allow-Credentials true always;
            add_header Access-Control-Allow-Methods "GET,POST,PUT,DELETE,PATCH,OPTIONS" always;
            add_header Access-Control-Allow-Headers "Authorization,Content-Type,X-Trace-Id,X-Client-Id,X-Device-Id,X-Timestamp,X-Nonce,X-Signature,X-App-Version,Idempotency-Key" always;
            add_header Access-Control-Max-Age 3600 always;
            add_header Vary Origin always;
            return 204;
        }

        # 实际请求
        add_header Access-Control-Allow-Origin $cors_origin always;
        add_header Access-Control-Allow-Credentials true always;
        add_header Access-Control-Expose-Headers "Authorization,X-Trace-Id" always;
        add_header Vary Origin always;

        proxy_pass http://your_backend_upstream; # 例如 http://127.0.0.1:8080
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
